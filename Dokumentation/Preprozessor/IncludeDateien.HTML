Include Dateien

In den beiliegenden Include- Dateien sind nachfolgende Macros vordefiniert:
DOCTYPE_def.inc
<!--#DEFINE HTML_4_Trans '<!DOCTYPE HTML PUBLIC "-//W3C//DTD HTML 4.0 Transitional//EN">' -->
<!--#DEFINE HTML_4_Frame '<!DOCTYPE HTML PUBLIC "-//W3C//DTD HTML 4.0 Frameset//EN">' -->
<!--#DEFINE HTML_4 '<!DOCTYPE HTML PUBLIC "-//W3C//DTD HTML 4.0//EN">' -->
<!--#DEFINE HTML_32 '<!DOCTYPE HTML PUBLIC "-//W3C//DTD HTML 3.2 Final//EN">' -->
<!--#DEFINE HTML_2 '<!DOCTYPE HTML PUBLIC "-//W3C//DTD HTML 2.0//EN">' -->
font_def.inc
<!--#DEFINE <FONT_RD> '<FONT COLOR=red>' -->
<!--#DEFINE <FONT_GN> '<FONT COLOR=green>' -->
<!--#DEFINE <FONT_BL> '<FONT COLOR=blue>' -->
<!--#DEFINE <FONT_YW> '<FONT COLOR=yellow>' -->
<!--#DEFINE <FONT_MN> '<FONT COLOR=maroon>' -->
<!--#DEFINE <FONT_TL> '<FONT COLOR=teal>' -->
<!--#DEFINE <FONT(v1)> '<FONT COLOR=v1>' -->
<!--#DEFINE <FONT+1> '<FONT SIZE="+1">' -->
<!--#DEFINE <FONT-1> '<FONT SIZE="-1">' -->
<!--#DEFINE <FONT(_v1)> '<FONT SIZE=_v1>' -->
<!--#DEFINE <FONT_SC(_v1,_v2)> '<FONT SIZE=_v1 COLOR=_v2>' -->
glob_def.inc
<!--#DEFINE _Home 'name.server.de/sonstwas' -->
<!--#DEFINE _Mail 'name@mail.net' -->
<!--#DEFINE _Autor 'Name' -->

Bitte die eigenen Daten einsetzen.
Die Macros werden in MetaVar_def.inc und std_def.inc weiterverwendet.
MetaVar_def.inc
<META HTTP-EQUIV="Content-Type" CONTENT="text/html;CHARSET=iso-8859-1">
<META HTTP-EQUIV="Erstellt" CONTENT="<!--#FILEDATE dd.mm.yyyy-->">
<META NAME="Content-Language" CONTENT="de">
<META NAME="generator" CONTENT="CompressHTML/HTML_Preprozessor">
<META NAME="author" CONTENT="_Autor">
<META NAME="publisher" CONTENT="_Autor">
<META NAME="copyright" CONTENT="_Autor">
std_def.inc
<!--#DEFINE <BODY_std> '<BODY BGCOLOR="#BBBBBB">' -->
<!--#DEFINE <H1_I> '<H1><I>' -->
<!--#DEFINE </H1_I> '</I></H1>' -->
<!--#DEFINE <DIV_C> '<DIV ALIGN=CENTER>' -->
<!--#DEFINE <DIV_R> '<DIV ALIGN=RIGHT>' -->
<!--#DEFINE <DIV_L> '<DIV ALIGN=LEFT>' -->
<!--#DEFINE <IMG(_v1,_v2,_v3)> '<IMG SRC="_v1" ALT="_v2" <!--#IMGSIZE '_v1' --> BORDER=_v3>' -->
<!--#DEFINE <Autor(_v1)>  '<DIV ALIGN=RIGHT>\n<FONT SIZE="-1"><I>>!--#FILEDATE dd. month yyyy-->, _v1</I></FONT>\n</DIV>' -->
<!--#DEFINE <std_Date> '<!--#FILEDATE dd. month yyyy -->' -->
<!--#DEFINE <MAIL(_v1)> '<A HREF="mailto:_Mail?subject=_v1">' -->
<!--#DEFINE <HOME> '<A HREF="http://_Home">' -->
table_def.inc
<!--#DEFINE <TABLE(v1,v2,v3,v4,v5)> '<TABLE WIDTH=v1 BORDER=v2 CELLPADDING=v3 CELLSPACING=v4 BGCOLOR=v5>' -->
<!--#DEFINE <TRow> '<TR>\n <TD>' -->
<!--#DEFINE </TRow> '</TD>\n</TR>\n' -->
<!--#DEFINE MARKER '</TD>\n <TD>' -->
<!--#DEFINE <TABLE_std> '<TABLE BORDER=0 WIDTH="90%" CELLSPACING=3 BGCOLOR="#CCCCCC">' -->
<!--#DEFINE <TD_VT> '<TD VALIGN=TOP>' -->
<!--#DEFINE <TD_AC> '<TD ALIGN=CENTER>' -->
<!--#DEFINE <TD_AR> '<TD ALIGN=RIGHT>' -->

Für diese Macros, bitte die Beispiel- Datei Tabelle.HTML ansehen.
